03 import java.io.IOException;
04 import java.io.ByteArrayOutputStream;
05 import java.io.PrintStream;
06 import java.util.Properties;
07 import examples.common.thirdparty.GenericResourceLoader;
08 import weblogic.utils.Debug;
09 
10 
11 /**
12  * AppUtils is a utility class that can be used by all modules in the ear
13  *
14  * @author Copyright (c) 2010, Oracle and/or its affiliates. All Rights Reserved.
15  */
16 public class AppUtils {
17   
18   private final static String APPLICATION_PROPERTIES_NAME = "applicationresource.properties";
19   private final static String EJB_JNDI_NAME_PROPERTIES_NAME = "hello.ejb.jndi.name";
20   
21   private static Properties appProperties;
22   private static boolean debug = true;
23 
24   private static AppUtils appUtils = null;
25 
26   /* static method to get AppUtils */
27   public static AppUtils getAppUtils() {
28     if (appUtils == null) {
29       throw new AssertionError("getAppUtils() called prior to appUtil's being initialized");
30     }
31     return appUtils;
32   }
33   
34   /* protect our constructor */
35   private AppUtils() {
36     try {
37       if (debug) {
38         Debug.say("loading appProperties");
39       }
40       appProperties = GenericResourceLoader.getProperties("/"+APPLICATION_PROPERTIES_NAME);
41       if (debug) {
42         Debug.say("appProperties loaded : " + appProperties);
43       }
44     } catch (IOException ioe) {
45       throw new IllegalStateException(throwable2StackTrace(ioe));
46     } 
47   }
48     
49   /**
50    * Typically the property related methods in this class they would be static's
51    * however to demonstrate a common pattern where a set of resources
52    * need to be loaded at application deployment, and reloaded when an app
53    * is redeployed we use this approach.  This method will be called by
54    * our applicaiton lifecycle listener, no one else should call this.
55    */
56   public static void initAppUtils() {
57     if (debug) {
58       Debug.say("AppUtils initAppUtils");
59     }
60     appUtils = new AppUtils();
61   }
62   
63   public Properties getProps() {
64     return appProperties;
65   }
66   
67   public String getProperty(String propertyName) {
68     return appProperties.getProperty(propertyName);
69   } 
70 
71   public String getEJBJNDIName() {
72     return getProperty(EJB_JNDI_NAME_PROPERTIES_NAME);
73   }
74 
75   public static String throwable2StackTrace(Throwable th) {
76     if(th == null) th = new Throwable("[Null exception passed, creating stack trace for offending caller]");
77     ByteArrayOutputStream ostr = new ByteArrayOutputStream();
78     th.printStackTrace(new PrintStream(ostr));
79     return ostr.toString();
80   }
81 }
