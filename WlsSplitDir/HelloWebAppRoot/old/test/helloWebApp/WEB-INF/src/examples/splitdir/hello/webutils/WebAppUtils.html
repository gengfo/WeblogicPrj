03 import javax.servlet.http.HttpServletRequest;
04 
05 
06 /**
07  * WebApppUtils is a utility class that will be used by just this single webapp
08  *
09  * If it was used by more than one webapp the source would live at 
10  * helloWorldEar/webutils/examples/splitdir/hello/webutils
11  * and would be automatically compiled into APP-INF/classes
12  *
13  * This is not a complete utility, you can imagine also wanting to get the 
14  * version number of the browser.
15  *
16  * @author Copyright (c) 2010 by BEA Systems, Inc. All Rights Reserved.
17  */
18 
19 public class WebAppUtils {
20   
21   private final static String USER_AGENT = "User-Agent";
22   //User-Agent: Mozilla/4.0 (compatible; MSIE 6.0; Windows NT 5.0)
23   //User-Agent: Mozilla/5.0 (Windows; U; Windows NT 5.0; en-US; rv:1.0.1) Gecko/20020823 Netscape/7.0
24   //User-Agent: Mozilla/4.7 [en] (WinNT; I)
25 
26   private final static String IE_INDEX = "MSIE";
27   private final static String NES_INDEX = "Mozilla";
28 
29   public final static int INTERNET_EXPLORER = 3;
30   public final static int NETSCAPE = 1;
31 
32   /**
33    * Not an exhaustive list by any means
34    */
35   private static int getBrowserType(HttpServletRequest req) {
36     String tempH = getHeader(req, USER_AGENT);
37     if(tempH != null && tempH.indexOf(IE_INDEX)!= -1) {
38       return 3;
39     } else if(tempH != null && (tempH.indexOf(NES_INDEX) != -1 && tempH.indexOf(IE_INDEX) == -1)) {
40       return 1;
41     }
42     return -1;
43   }
44   /**
45    * Utility method to get the vendor name of the browser
46    *
47    * Again this is not a full list, just a sample.
48    */
49 
50   public static String getBrowserName(HttpServletRequest req) {
51     switch(getBrowserType(req)) {
52     case INTERNET_EXPLORER:
53       return "Internet Explorer";
54     case NETSCAPE:
55       return "Netscape";
56     default:
57       return "Unknown";
58     }
59   }
60 
61   private static String getHeader(HttpServletRequest req, String name) {
62     return req.getHeader(name);
63   }
64 
65 }
